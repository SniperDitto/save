        ,O_EMSG     OUT     VARCHAR2

CREATE OR REPLACE
PACKAGE BODY PKG_ADMIN_AUTH AS

  PROCEDURE PROC_SEL_AUTH
    (
        IN_AUTHNAME     IN      VARCHAR2
        ,O_CUR          OUT     SYS_REFCURSOR
    ) AS
  BEGIN
    
    OPEN O_CUR FOR
    SELECT AUTHID, AUTHNAME
    FROM AUTHES
    WHERE AUTHNAME LIKE '%'|| IN_AUTHNAME ||'%'
    ORDER BY AUTHID ASC
    ;
    

  END PROC_SEL_AUTH;
  
  
  
  
  
  PROCEDURE PROC_SAVE_AUTH
