CREATE OR REPLACE
PACKAGE BODY PKG_PRODUCT AS

  PROCEDURE PROC_INS_PRODUCT
    (
        IN_PNAME    IN  VARCHAR2
        ,IN_PPRICE  IN  NUMBER
    ) AS
        V_NEW_PID   CHAR(6);
  BEGIN
    SELECT 'P'||TO_CHAR(TO_NUMBER(SUBSTR(NVL(MAX(PID),'P00000'),2,5))+1,'FM00000')
    INTO V_NEW_PID
    FROM PRODUCTS;

    INSERT INTO PRODUCTS(PID, PNAME, PPRICE) 
    VALUES(V_NEW_PID, IN_PNAME, IN_PPRICE);

  END PROC_INS_PRODUCT;

END PKG_PRODUCT;
