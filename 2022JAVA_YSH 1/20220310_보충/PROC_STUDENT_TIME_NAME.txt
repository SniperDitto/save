

    CREATE OR REPLACE PROCEDURE PROC_STUDENT_TIME_NAME 
    (
      IN_START_DATE IN VARCHAR2 
    , IN_FINISH_DATE IN VARCHAR2 
    , O_CUR OUT SYS_REFCURSOR
    ) AS 
    BEGIN
        OPEN O_CUR FOR
        SELECT T1.STU_ID, T1.STU_NAME
        FROM STUDENTS_TBL T1, STUDENTS_TIME_TBL T2
        WHERE T1.STU_ID=T2.STU_ID
            AND TO_CHAR(T2.REG_TIME, 'YYYYMMDD') >= IN_START_DATE
            AND TO_CHAR(T2.REG_TIME, 'YYYYMMDD') <= IN_FINISH_DATE
        GROUP BY T1.STU_ID, T1.STU_NAME
        ;
    END PROC_STUDENT_TIME_NAME;
